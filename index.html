SQL Setup Supabase ✕

Jalankan SQL ini di Supabase → SQL Editor:

-- Jalankan SQL ini di Supabase → SQL Editor

-- Tabel config
CREATE TABLE IF NOT EXISTS pam_config (
  key TEXT PRIMARY KEY,
  value JSONB NOT NULL
);

-- Tabel warga
CREATE TABLE IF NOT EXISTS pam_warga (
  id TEXT PRIMARY KEY,
  nama TEXT NOT NULL,
  alamat TEXT DEFAULT '',
  telp TEXT DEFAULT '',
  tarif INTEGER DEFAULT 2000,
  meter_awal INTEGER DEFAULT 0,
  status TEXT DEFAULT 'aktif',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Tabel tagihan
CREATE TABLE IF NOT EXISTS pam_tagihan (
  id TEXT PRIMARY KEY,
  warga_id TEXT REFERENCES pam_warga(id) ON DELETE CASCADE,
  periode TEXT NOT NULL,
  meter_awal NUMERIC DEFAULT 0,
  meter_akhir NUMERIC DEFAULT 0,
  pemakaian NUMERIC DEFAULT 0,
  tarif INTEGER DEFAULT 2000,
  abonemen INTEGER DEFAULT 5000,
  total NUMERIC DEFAULT 0,
  status TEXT DEFAULT 'belum',
  tgl_bayar TEXT,
  catatan TEXT DEFAULT '',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Aktifkan Row Level Security (opsional, sesuaikan dengan kebutuhan)
ALTER TABLE pam_config ENABLE ROW LEVEL SECURITY;
ALTER TABLE pam_warga ENABLE ROW LEVEL SECURITY;
ALTER TABLE pam_tagihan ENABLE ROW LEVEL SECURITY;

-- Policy: izinkan semua operasi dari anon key (untuk single-user app)
CREATE POLICY "allow_all_config" ON pam_config FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY "allow_all_warga" ON pam_warga FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY "allow_all_tagihan" ON pam_tagihan FOR ALL TO anon USING (true) WITH CHECK (true);
